\def\trim#1{%
    \ignorespaces#1\unskip%
}

\def\SECTION#1{%
    % \section*{\normalfont \textbf{#1}}%
    \specialsection*{\textbf{#1}}%
}

\def\tline#1#2{%
    $\underset{\text{#1}}{\text{\underline{\hskip #2}}}$%
}

\def\sign#1{%
    \tline{(подпись)}{100pt}
    {\hyphenchar\font=-1 /~\trim{#1}~/}%
}

\def\dateRangeCell#1#2{%
    \makecell[lt]{#1\,-- \\ #2}%
}

\newlength\fieldwidth

\def\fullLineField#1#2{%
    \def\filedname{#1\quad}%
    \setlength{\fieldwidth}{\textwidth - \widthof{\filedname}}%
    \filedname \underline{\makebox[\fieldwidth][c]{#2}}%
}

\def\multiLineField#1#2{%
    \def\filedname{#1\quad}%
    \setlength{\fieldwidth}{\textwidth - \widthof{\filedname}}%
    \filedname
    % \begin{minipage}[t]{\fieldwidth}
        \ul#2%
    % \end{minipage}
}
